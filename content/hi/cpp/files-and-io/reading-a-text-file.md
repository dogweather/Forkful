## How to: (कैसे करें:)
```C++
#include <iostream>
#include <fstream>
#include <string>

int main() {
    std::ifstream inputFile("example.txt");
    std::string line;

    if (inputFile.is_open()) {
        while (getline(inputFile, line)) {
            std::cout << line << '\n';
        }
        inputFile.close();
    } else {
        std::cout << "फाइल खोलने में असमर्थ।" << std::endl;
    }

    return 0;
}
```
