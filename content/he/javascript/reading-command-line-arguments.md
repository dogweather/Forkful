# למה

למה לקרוא ארגומנטים מקו הפקודה? למידה על ארגומנטים מקו הפקודה יכולה לעזור לשפר את יכולת התכנות שלכם ולהפוך את העבודה שלכם עם סקריפטים של Javascript לקלה יותר ויעילה.

# איך לעשות זאת

בפוסט הזה, אני אראה לכם כיצד לקרוא ולהשתמש בארגומנטים מקו הפקודה בסקריפטים של Javascript. בכדי להראות לכם את כל האיפשרויות, אני אשתמש במקרים שונים בכדי להדגים איך אפשר למצוא ולהשתמש בארגומנטים האלה.

```Javascript
// דוגמא 1: קבלת ארגומנטים מקו הפקודה
node script.js hello world
// תוצאה: הארגומנטים החדשים "hello" ו-"world" אותרו במערך הקבוע process.argv
// הסיקור נראה כך:
[ '', '', 'hello', 'world' ]

// דוגמא 2: אינדקסים באותו הקשר
node script.js -x -i input.txt -o output.txt
// תוצאה: האינדקסים עובדים במקרה של אפשרויות כמו נתיבי קלט ופלט עבור התוכנית
// הסיקור נראה כך:
[ '', '', '-x', '-i', 'input.txt', '-o', 'output.txt' ]

// דוגמא 3: פריסה עבור ארגומנטים
var args = process.argv.slice(2);
node script.js -a arg1 arg2 arg3
// תוצאה: קבלת מערך עם רק נתיב לקבלת הארגומנטים, מתחילים מהאינדקס שנתחיל לכתוב, כך כו...
// הסיקור נראה כך:
[ '-a', 'arg1', 'arg2', 'arg3' ]
```
