# למה:
פעמים רבות, כאשר אנו עובדים עם נתונים בתוכניות ארדואינו, אנו מתנקזים על העיבוד והטיפול בתוך הבקשות והתנאים המסורבלים. בעזרת תכנון אינטליגנטי של ביטויים רגילים, בקיצור "regular expressions", אנחנו יכולים להפוך את הכתיבה וקריאת הקוד שלנו לפשוטה יותר ולחסוך זמן יקר.

## איך לעשות זאת:
בשפת ארדואינו, ישנן מספר ספריות שמאפשרות לנו להשתמש ברגקספ כדי למצוא ולעבד מידע בצורה יעילה יותר. לדוגמא, הספריה "rexexp" מאפשרת לנו להשתמש בפונקציות כמו "match" ו-"find" כדי לחפש מידע מסוים בתוך מחרוזות. הנה דוגמאות לשימוש בביטויי רגילים בארדואינו:

```Arduino
#include <regex.h>

void setup() {
  Serial.begin(9600);
  String text = "הנה מילה כלשהי בתוך הטקסט";
  
  // חיפוש מילה כלשהי בתוך הטקסט
  if (regexMatch(text, "מילה")) {
    Serial.println("מצאנו את המילה!");
  }
  
  // החלפת מילה כלשהי בתוך הטקסט
  text = regexReplace(text, "כלשהי", "חדשה");
  Serial.println("הטקסט המעודכן: " + text);
  
  // חיפוש מספר בתוך הטקסט
  if (regexMatch(text, "\\d+")) { // מצא 1 או יותר ספרות
    Serial.println("מצאנו מספר בטקסט!");
  }
  
  // חיפוש דוא"ל בתוך הטקסט
  if (regexMatch(text, "[a-z0-9._%+-]+@[a-z0-9.-]+\\.[a-z]{2,}")) {
    Serial.println("מצאנו כתובת דוא\"ל בטקסט!");
  }
}

void loop() {
  // קוד תוכנית ארדואינו רגיל כאן...
}
```
