```C#
// מקבל ארגומנטים משורת הפקודה
static void Main(string[] args)
{
    if(args.Length > 0)
    {
        Console.WriteLine("הנה הארגומנטים שקיבלת:");
        foreach (var arg in args)
        {
            Console.WriteLine(arg);
        }
    }
    else
    {
        Console.WriteLine("לא קיבלתי ארגומנטים כלל. תנסה שוב.");
    }
}
```
